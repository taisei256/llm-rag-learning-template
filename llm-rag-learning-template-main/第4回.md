```javascript
// 簡単な実装例  
function recommendNextTopic(learningHistory, ontology) {
    // 学習済み概念から、次に学ぶべき概念を推薦
    const learned = learningHistory.map(h => h.concept);
    const nextConcepts = [];
    
    Object.entries(ontology.concepts).forEach(([id, concept]) => {
        const prerequisitesMet = concept.prerequisites.every(p => 
            learned.includes(p)
        );
        if (prerequisitesMet && !learned.includes(id)) {
            nextConcepts.push(concept);
        }
    });
    
    return nextConcepts[0]; // 最初の候補を返す
}
```

**できること**:
- 学習履歴から次の学習内容を提案
- オントロジーの前提知識を考慮
